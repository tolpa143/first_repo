cd {папка} - путь к папке
git init - делает выбранную директорию репозиторием
git remote add origin {url} - локальный репозиторий - удаленный репозиторий
git clone {url} - копирование удаленного репозитория
git status - статус коммитов
git add . - коммит всех файлов(индексирование)
git commit -m "Сообщение" - запоминает коммит, присваивает хэш
git log - информация о коммитах
git checkout {хэш} - возвращение к коммиту
[master - имя основной ветки]
[origin  - путь файла?]
git push origin master- публикует коммиты в удаленный реп
git pull origin master - скачивание из удаленного репо все коммиты
git branch - список локальных веток
git branch {имя ветки} - создает новую ветку
git checkout {имя ветки} - переход на другую ветку
git merge {имя ветки} - слиянеие веток (коммит становится один)
git rebase {имя ветки} - слиянеие веток (все коммиты сохраняются)
